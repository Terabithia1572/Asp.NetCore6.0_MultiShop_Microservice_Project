    <script>
        // Profile dropdown with fade/slide
        const btn = document.getElementById('profileBtn');
        const menu = document.getElementById('profileMenu');
        btn.addEventListener('click', () => menu.classList.toggle('show'));
        window.addEventListener('click', e => { if (!btn.contains(e.target)) menu.classList.remove('show'); });

        // Animated counters
        document.querySelectorAll('.counter').forEach(el => {
            const to = parseFloat(el.dataset.to);
            const isFloat = String(to).includes('.');
            let cur = 0;
            const steps = 40, inc = to / steps;
            const t = setInterval(() => {
                cur += inc;
                el.textContent = isFloat ? cur.toFixed(1) : Math.floor(cur);
                if (cur >= to) { el.textContent = isFloat ? to.toFixed(1) : to; clearInterval(t); }
            }, 20);
        });

        // Big sales chart
        const ctx = document.getElementById('salesChart').getContext('2d');
        new Chart(ctx, {
            type: 'bar',
            data: {
                labels: ['Ocak', 'Şubat', 'Mart', 'Nisan', 'Mayıs', 'Haziran'],
                datasets: [{ data: [120, 160, 190, 250, 220, 310], backgroundColor: 'rgba(255,193,7,0.9)', borderRadius: 7 }]
            },
            options: { plugins: { legend: { display: false } }, scales: { y: { beginAtZero: true, grid: { color: '#eee' } }, x: { grid: { display: false } } } }
        });

        // Mini visits line chart
        const vctx = document.getElementById('visitChart').getContext('2d');
        new Chart(vctx, {
            type: 'line',
            data: {
                labels: ['Pzt', 'Sal', 'Çar', 'Per', 'Cum', 'Cmt', 'Paz'],
                datasets: [{
                    data: [180, 210, 195, 230, 260, 240, 270], tension: .35, fill: true,
                    backgroundColor: 'rgba(255,193,7,.18)', borderColor: '#ffc107', pointRadius: 0, borderWidth: 2
                }]
            },
            options: { plugins: { legend: { display: false } }, scales: { y: { beginAtZero: true, grid: { color: '#f0f0f0' } }, x: { grid: { display: false } } } }
        });

        // Live clock
        const clock = document.getElementById('clock');
        const tick = () => { const d = new Date(); clock.textContent = d.toLocaleTimeString('tr-TR', { hour: '2-digit', minute: '2-digit', second: '2-digit' }); };
        tick(); setInterval(tick, 1000);
    </script>
